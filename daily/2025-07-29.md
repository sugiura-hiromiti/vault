const today = '2025-07-29'
dv.list(
	dv.pages()
		.filter(p => {
			const date = p.file.mday;
			const year = `${date.c.year}`;
			const month = `${date.c.month + 100}`.slice(1, 3);
			const day = `${date.c.day + 100}`.slice(1, 3);

			return today === formatted_today;
		})
