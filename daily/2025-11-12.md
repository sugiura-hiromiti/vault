function __self_then_complete
    # Insert the typed character, then trigger the completion popup
    commandline -f self-insert
    commandline -f complete
end

function fish_user_key_bindings
    # Bind ALL printable ASCII in insert mode to our wrapper
    for i in (seq 32 126)
        set -l i (printf "%02x" $i)
        set -l key (printf "\x$i")
        echo "key is $key"
        bind -M insert $key __self_then_complete
    end

    # Keep Tab/Enter behaving normally (not strictly necessary; shown for clarity)
    bind -M insert \t complete
    bind -M insert \n execute

    # Optional: make Backspace also re-compute completions
    function __backspace_then_complete
        commandline -f backward-delete-char
        commandline -f complete
    end
    bind -M insert \b __backspace_then_complete
end

bind ---sets-mode insert
